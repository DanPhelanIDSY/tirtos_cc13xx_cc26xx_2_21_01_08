214 /* @brief Fast/efficient version of #PIN_setOutputEnable()
215  * @note Does not include any checks on handle for efficiency reasons,
216  * use #PIN_setOutputEnable() for checked version
217  */
218 __STATIC_INLINE void PINCC26XX_setOutputEnable(PIN_Id pinId, bool bOutEn) {
219  uint32_t key = Hwi_disable();
220  HWREG(GPIO_BASE+GPIO_O_DOE31_0) =
221  ((HWREG(GPIO_BASE+GPIO_O_DOE31_0) & ~(1<<pinId)) | (bOutEn<<pinId));
222  Hwi_restore(key);
223 }

275 __STATIC_INLINE void PINCC26XX_setPortOutputEnable(PIN_Handle handle, uint_t bmOutEn) {
276  // Only a single port on CC26xx
277  uint32_t key = Hwi_disable();
278  HWREG(GPIO_BASE+GPIO_O_DOE31_0) =
279  (HWREG(GPIO_BASE+GPIO_O_DOE31_0) & (~handle->bmPort)) | (bmOutEn & handle->bmPort);
280  Hwi_restore(key);
