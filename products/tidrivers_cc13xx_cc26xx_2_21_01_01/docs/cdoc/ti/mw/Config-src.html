    37    /*!
    38     *  ======== Config ========
    39     */
    40    metaonly module Config {
    41    
    42        /*!
    43         *  ======== provideDisplayLib ========
    44         *  Adds the Display middleware library to the project.
    45         *
    46         *  This option is only supported on CC13XX & CC26XX devices and the
    47         *  default value is true.
    48         *
    49         */
    50        metaonly config Bool provideDisplayLib;
    51    
    52        /*!
    53         *  ======== provideGrLib ========
    54         *  Adds the GrLib middleware library to the project.
    55         *
    56         *  This option is only supported on CC13XX & CC26XX devices and the
    57         *  default value is true.
    58         *
    59         */
    60        metaonly config Bool provideGrLib;
    61    
    62        /*!
    63         *  ======== provideLCDLib ========
    64         *  Adds the LCD middleware library to the project.
    65         *
    66         *  This option is only supported on CC13XX & CC26XX devices and the
    67         *  default value is true.
    68         *
    69         */
    70        metaonly config Bool provideLCDLib;
    71    
    72        /*!
    73         *  ======== provideWiFiCC3X00Lib ========
    74         *  Adds the CC3X00 WiFi host driver middleware library & the TI-RTOS WiFi
    75         *  driver library to the project.
    76         *
    77         *  This option is only supported on MSP430, MSP432 & TivaC devices and the
    78         *  default value is true.
    79         *
    80         */
    81        metaonly config Bool provideWiFiCC3X00Lib;
    82    
    83        /*!
    84         *  ======== supportsDisplayLib ========
    85         *  @_nodoc
    86         */
    87        metaonly config Bool supportsDisplayLib;
    88    
    89        /*!
    90         *  ======== supportsGrLib ========
    91         *  @_nodoc
    92         */
    93        metaonly config Bool supportsGrLib;
    94    
    95        /*!
    96         *  ======== supportsLCDLib ========
    97         *  @_nodoc
    98         */
    99        metaonly config Bool supportsLCDLib;
   100    
   101        /*!
   102         *  ======== supportsWiFiCC3X00Lib ========
   103         *  @_nodoc
   104         */
   105        metaonly config Bool supportsWiFiCC3X00Lib;
   106    }
