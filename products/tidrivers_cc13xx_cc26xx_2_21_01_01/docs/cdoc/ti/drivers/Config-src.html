    33    /*
    34     *  ======== Config.xdc ========
    35     *  This module is used to tie together the xGConf page for TI Drivers product.
    36     */
    37    
    38    /*!
    39     *  ======== Config ========
    40     */
    41    
    42    metaonly module Config {
    43        /*!
    44         *  ======== LibType ========
    45         *  Configuration options for TI Drivers library
    46         *
    47         *  You can select the library type by setting
    48         *  the {@link #libType Config.libType} configuration parameter.
    49         *
    50         *  @field(LibType_Instrumented) The library supplied is pre-built with
    51         *  logging and assertions enabled.
    52         *  Diags_USER1 mask logs basic information
    53         *  Diags_USER2 mask logs more detailed information
    54         *
    55         *  @field(LibType_NonInstrumented) The library supplied is pre-built
    56         *  with logging and assertions disabled.
    57         */
    58        enum LibType {
    59            LibType_Instrumented,           /*! instrumented (with asserts and logs)*/
    60            LibType_NonInstrumented         /*! non-instrumented */
    61        };
    62    
    63        /*!
    64         *  ======== libType ========
    65         *  TI Drivers library type
    66         *
    67         *  The driver libraries are provided to link with your application.
    68         *  Two forms of this library are provided with the TI Drivers product.
    69         *  This configuration parameter allows you to select the form of the
    70         *  driver libraries to use.
    71         *
    72         *  The default value of libType is
    73         *  {@link #LibType_NonInstrumented LibType_NonInstrumented}.  For a
    74         *  complete list of options and what they offer see {@link #LibType}.
    75         */
    76        metaonly config LibType libType = LibType_NonInstrumented;
    77    
    78        /*!
    79         *  ======== RFDriverMode ========
    80         *  TI-Drivers RF Driver mode
    81         *
    82         *  You can select the RF driver mode by setting the
    83         *  {@link #rfDriverMode Config.rfDriverMode} configuration parameter.
    84         *
    85         *  @field(RF_SingleMode) A single client/protocol will have exclusive
    86         *  access to the radio.
    87         *
    88         *  @field(RF_MultiMode) Supports multiple clients/protocols accessing
    89         *  the radio simultaneously.
    90         */
    91        enum RFDriverMode {
    92            RF_MultiMode,        /*! Multiple clients/protocols run simultaneously */
    93            RF_SingleMode        /*! Only a single client/protocol */
    94        };
    95    
    96        /*!
    97         *  ======== rfDriverMode ========
    98         *  TI-Drivers RF Driver mode
    99         *
   100         *  The RF driver supports single & multi modes of operation.  In
   101         *  single-mode only a single client/protocol can interact  with the
   102         *  radio, meaning if another protocol is required, the RF driver must be
   103         *  closed & re-opened.
   104         *
   105         *  In multi-mode more than one client/protocol can interact with the radio
   106         *  simultaneously.
   107         *
   108         *  The default value of rfDriverMode is:
   109         *  {@link #RF_SingleMode RF_SingleMode}.  For a
   110         *  complete list of options and what they offer see {@link #RFDriverMode}.
   111         */
   112        metaonly config RFDriverMode rfDriverMode = RF_SingleMode;
   113    
   114        /*!
   115         *  ======== supportsRFDriver ========
   116         *  @_nodoc
   117         */
   118        metaonly config Bool supportsRFDriver = false;
   119    
   120        /*!
   121         *  ======== useWiFiDriver ========
   122         *  @_nodoc
   123         *  This variable can be used to add the WiFi driver library is during
   124         *  linking.  Default value is true, but may be changed by the ti.mw.Config
   125         *  module.
   126         *
   127         */
   128        metaonly config Bool useWiFiDriver = true;
   129    }
